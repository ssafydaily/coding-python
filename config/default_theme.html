# Config

navbar

  • 타입: false | (NavbarItem | NavbarGroup | strirg)[]

  • 기본값: []

  • false: navbar 사용안함

export default {
  theme: defaultTheme({
    navbar: [
      // NavbarItem
      {
        text: 'Foo',
        link: '/foo/',
      },
      // NavbarGroup
      {
        text: 'Group',
        children: ['/group/foo.md', '/group/bar.md'],
      },
      // string - page file path
      '/bar/README.md',
    ],
  }),
}
